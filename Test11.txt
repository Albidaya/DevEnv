SELECT  fct.prod_key,fct.stat_unit_qty su,fct.gross_transact_amt giv FROM ship_fct FCT
        Join cust_dim_ext cust2 on cust2.cust_key = fct.sold_to_cust_key
        JOIN prod_dim_ext prod ON (fct.prod_key = prod.prod_key and prod.attr_data_provider_key = cust2.attr_data_provider_key)
WHERE   1=1
        AND time_period_end_date >= date_add(last_day(add_months(current_date, -7)),1)
        AND cust2.attr_data_provider_key IN (   10000000000004, 
                                                10000000000003, 
                                                10000000000008, 
                                                10000000000009, 
                                                10000000000007, 
                                                10000000000006, 
                                                10000000000005,
                                                10000000000010)
        AND prod.attr_data_provider_key IN (    10000000000004, 
                                                10000000000003, 
                                                10000000000008, 
                                                10000000000009, 
                                                10000000000007, 
                                                10000000000006, 
                                                10000000000005,
                                                10000000000010);
